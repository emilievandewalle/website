  const colors = [

    { bg: '#E6D3B1', text: '#2B2B2B' },

    { bg: '#9FB3C1', text: '#1F1F1F' },

  ];


  function setRandomColor() {

    const c = colors[Math.floor(Math.random() * colors.length)];

    document.body.style.backgroundColor = c.bg;

    document.body.style.color = c.text;

  }

  });
